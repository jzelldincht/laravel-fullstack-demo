# Api 模块异常处理
我们以模块`Api`为例，接管应用和`Api`模块的异常处理。实现自定义异常处理的机制。

## 1. 在`bootstrap/app.php`文件新增一个异常处理类
```php 
// 设置modules/Common/下的异常处理类，用于处理所有模块的异常
$app->singleton(
    Illuminate\Contracts\Debug\ExceptionHandler::class,
    Modules\Common\Exceptions\Handler::class
);
```

## 2. 新增modules/common/Exceptions/Handler.php

复制`app/Exceptions/Handler.php`到`modules/common/Exceptions/Handler.php`
```php 
## modules/common/Exceptions/Handler.php

<?php

namespace Modules\Common\Exceptions;

use Illuminate\Foundation\Exceptions\Handler as ExceptionHandler;
use Illuminate\Validation\ValidationException;
use Modules\Api\Exceptions\ApiException;
use Throwable;

class Handler extends ExceptionHandler
{
    /**
     * A list of the exception types that are not reported.
     *
     * @var array<int, class-string<Throwable>>
     */
    protected $dontReport = [
        //
    ];

    /**
     * A list of the inputs that are never flashed for validation exceptions.
     *
     * @var array<int, string>
     */
    protected $dontFlash = [
        'current_password',
        'password',
        'password_confirmation',
    ];

    /**
     * Register the exception handling callbacks for the application.
     *
     * @return void
     */
    public function register()
    {
        $this->reportable(function (Throwable $e) {
            //
        });
    }

    public function render($request, Throwable $e)
    {
        // 判断是否是 api/xx/xx  的路由
        if($request->is('api/*')) {
            if($e instanceof ApiException) {
                $message = $e->getMessage();
                $status = $e->getCode();

                return response()->json([
                    'status' => $status,
                    'message' => $message,
                ], $status);
            }

            if($e instanceof ValidationException)
            {
                $message = $e->validator->errors()->first();
                $status = 400;

                return response()->json([
                    'status' => $status,
                    'message' => $message,
                ], $status);
            }
        }


        return parent::render($request, $e); // TODO: Change the autogenerated stub
    }
}
```

```php 
namespace Modules\Api\Exceptions;

use Throwable;

class ApiException extends \Exception
{
    /**
     * Api异常接管
     * @param array $errdata [status => httpCode, message => error information]
     * @param Throwable|null $previous
     */
    public function __construct(array $errdata = ['status' => 500, 'message' => 'An error has occured processing your request.'], Throwable $previous = null)
    {
        parent::__construct($errdata['message'], $errdata['status'], $previous);
    }
}

```
